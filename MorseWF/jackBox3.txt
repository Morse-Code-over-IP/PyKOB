// Jack Box

#define jack1Pin 8
#define jack2Pin 10
#define jack3Pin 12
#define ledPin 13
int ledState = LOW;
int jackState;

void setup()
{
  pinMode(ledPin, OUTPUT);
  pinMode(jack1Pin, INPUT_PULLUP);
  pinMode(jack2Pin, INPUT_PULLUP);
  pinMode(jack3Pin, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop()
{
  jackState = 0x40;
  if (digitalRead(jack1Pin) == LOW)
    jackState |= 1;
  if (digitalRead(jack2Pin) == LOW)
    jackState |= 2;
  if (digitalRead(jack3Pin) == LOW)
    jackState |= 4;
  Serial.write(jackState);
  if (ledState == LOW)
  {
    digitalWrite(ledPin, HIGH);
    ledState = HIGH;
  }
  else
  {
    digitalWrite(ledPin, LOW);
    ledState = LOW;
  }
  delay(500);
}
